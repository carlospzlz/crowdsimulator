517 {

525  yRot = atan2(-_vector.m_z,_vector.m_x) * 180/M_PI;
526

539  //SENSE

650 void GLWindow::mouseMoveEvent(QMouseEvent * _event)
651 {
652  bool rotate;
653 
654  if( (rotate = _event->buttons()==Qt::LeftButton && m_rotate ) ||
655  (_event->buttons() == Qt::RightButton && m_translate ) )
656  {
657  if (rotate)
658  {
659  m_globalRotation.m_x += s_rotationIncrement * (_event->y() - m_previousMousePosition.second);
660  m_globalRotation.m_y += s_rotationIncrement * (_event->x() - m_previousMousePosition.first);
661  m_previousMousePosition.first = _event->x();
662  m_previousMousePosition.second = _event->y();
663  }
664  else
665  {
666  m_globalTranslation.m_x += s_translationIncrement * (_event->x() - m_previousMousePosition.first);
667  m_globalTranslation.m_y -= s_translationIncrement * (_event->y() - m_previousMousePosition.second);
668  m_previousMousePosition.first = _event->x();
669  m_previousMousePosition.second = _event->y();
670 
671  }
672  ngl::Transformation globalTransform;
673  ngl::Mat4 xRotMat, yRotMat, globalMatrix;
674  xRotMat.rotateX(m_globalRotation.m_x);
675  yRotMat.rotateY(m_globalRotation.m_y);
676  globalMatrix = yRotMat*xRotMat;
677  globalMatrix.m_m[3][0] = m_globalTranslation.m_x;
678  globalMatrix.m_m[3][1] = m_globalTranslation.m_y;
679  globalMatrix.m_m[3][2] = m_globalTranslation.m_z;
680  globalTransform.setMatrix(globalMatrix);
681  m_transformStack.setGlobal(globalTransform);
682 
683  updateGL();
684  }
